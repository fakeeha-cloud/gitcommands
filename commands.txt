git config --global user.name "your user name"        [to set username]

git config --global user.email "your email"           [to set email]

git init                                              [to initialize git]

git status                                            [to list untracked files]

git add -A                                            [add all untracked files to staging area]

git commit -m "commit message"                        [to move staged file to local repository]

git branch                                            [list all branches]

git remote add remote_name remote_path                [add remote repository to local repository]

git push remote_name branch_name                      [push to remote repository(github)]

git remote -v                                         [to list all remote_name]
